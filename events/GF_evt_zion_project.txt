		}
	}
	desc = {
		text = GF_zion_project.1000.desc.repeat
		trigger = {
			has_country_flag = flag_GF_requested_elid_army
			NOT = { has_valid_civic = civic_reanimated_armies }
		}
	}
	desc = {
		text = GF_zion_project.1000.desc.reanimated_armies
		trigger = { has_valid_civic = civic_reanimated_armies }
	}

	option = {
		name = GF_zion_project.1000.a
		is_dialog_only = yes
		response_text = GF_zion_project.1000.a.response
	}

	option = {
		name = GF_zion_project.1000.b
		is_dialog_only = yes
		response_text = GF_zion_project.1000.b.response
	}

	option = {
		name = GF_zion_project.1000.request
		response_text = GF_zion_project.1000.request.response

		trigger = {
			NOR = {
				has_country_flag = flag_GF_requested_elid_army
				has_valid_civic = civic_reanimated_armies
			}
		}

		hidden_effect = {
			set_country_flag = flag_GF_requested_elid_army	# 标记
		}
		GF_enable_zion_project = { PROJECT = SP_GF_ZION_ELID_ARMY_PROJECT }
	}

	option = {
		name = GF_zion_project.1000.request
		response_text = GF_zion_project.1000.request.reanimated_armies_response

		trigger = {
			has_valid_civic = civic_reanimated_armies
			NOR = {
				has_tech_option = GF_tech_elid_army
				has_technology = GF_tech_elid_army
			}
		}

		hidden_effect = {
			set_country_flag = flag_GF_requested_elid_army	# 标记
		}
		add_research_option = GF_tech_elid_army
	}

	option = {
		name = GF_zion_project.return
		trigger = {
			has_country_flag = flag_GF_requested_elid_army
		}

		hidden_effect = {

		}

country_event = {
	id = GF_zion_project.32204
	title = GF_ZION_DEEP_DIVE
	desc = GF_zion_project.32204.desc
	picture = GFX_evt_GF_white_net

	is_triggered_only = yes

	immediate = {
		event_target:choosing_leader = {
			export_trigger_value_to_variable = {
				trigger = has_skill
				variable = current_skill
			}
			# 读取当前领袖的等级上限
			add_skill = 100
			export_trigger_value_to_variable = {
				trigger = has_skill
				variable = max_skill
			}
			set_skill = current_skill

			if = {
				limit = {
					check_variable = {
						which = current_skill
						value >= max_skill
					}
				}
				set_leader_flag = skill_at_max
			}
			# 添加的经验的计算公式:
			# skill <  10: -40 * skill + 900
			# skill >= 10: -40 *    10 + 900
			if = {
				limit = {
					check_variable = {
						which = current_skill
						value >= 10
					}
				}
				set_variable = {
					which = current_skill
					value = 10
				}
			}
			multiply_variable = {
				which = current_skill
				value = -40
			}
			change_variable = {
				which = current_skill
				value = 900
			}
		}
	}

	option = {
		name = GF_zion_project.32204.a
		event_target:choosing_leader = {
			if = {
				# 如果领袖已经到达最大等级且未拥有特殊特质，则给予特殊特质
				limit = {
					has_leader_flag = skill_at_max
					NOR = {
						has_trait = GF_trait_leader_inspired_governor
						has_trait = GF_trait_leader_inspired_scientist
						has_trait = GF_trait_leader_inspired_admiral
						has_trait = GF_trait_leader_inspired_general
					}
				}
				switch = {
					trigger = leader_class
					governor = { add_trait = GF_trait_leader_inspired_governor }
					scientist = { add_trait = GF_trait_leader_inspired_scientist }
					admiral = { add_trait = GF_trait_leader_inspired_admiral }
					general = { add_trait = GF_trait_leader_inspired_general }
				}
			}
			else = {
				event_target:choosing_leader = {
					add_experience = current_skill
				}
			}
		}
		GF_set_zion_deep_dive_cooldown = yes
	}

	after = {
		hidden_effect = {
			event_target:choosing_leader = {
				remove_leader_flag = skill_at_max
			}
		}
	}
}
