#跨星系攻击的星球列表，会添加检查过该星系的flag。移除该flag
remove_flag_sfdoll_planet_killer_target_effect = {
	optimize_memory
	hidden_effect = {
		every_system = {
			limit = {
				has_star_flag = flag_sfdoll_planet_killer_target
			}
			remove_star_flag = flag_sfdoll_planet_killer_target
		}
	}
}
