# 事件0:前置隐藏事件触发器，20年后母星触发开始事件
country_event = {

	is_triggered_only = yes
	trigger = {

			id = GF_sf.1
			days = 7200
		}
	}
}

#事件1:开局20年触发的事件，定位在母星
country_event = {
	id = GF_sf.1
	title = "GF_sf.1.name"
	desc = "GF_sf.1.desc"
	picture = GFX_evt_announcement
	show_sound = event_activating_unknown_technology
	location = event_target:GF_home_world
	trackable = yes
	is_triggered_only = yes

	immediate = {
		event_target:GF_home_world.solar_system = {
			set_star_flag = flag_GF_sf_event_system
		}
	}

	option = {
		name = GF_sf.1.a
		begin_event_chain = {
			event_chain = "GF_sf_event_chain"
			target = this
		}
		
		event_target:GF_home_world = {
			enable_special_project = {
				name = "GF_sf_event_project_1"
				location = this
				owner = root
			}
		}
	}
}

#事件2:在荒芜星球建立特殊项目2
planet_event = {
	id = GF_sf.2
	title = "GF_sf.2.name"
	desc = "GF_sf.2.desc"
	picture = GFX_evt_mysterious_signal
	show_sound = event_activating_unknown_technology
	location = event_target:GF_sf_site_target_2
	trackable = yes
	is_triggered_only = yes

	immediate = {
		closest_system = {
			limit = {
				NOT = { has_star_flag = flag_GF_sf_event_system }
				any_system_planet = { is_planet_class = pc_barren }
				OR = {
					NOT = { exists = space_owner }
					AND = {
						exists = space_owner
						space_owner = { is_same_value = root.owner }
					}
				}
			}
			min_steps = 2
			max_steps = 10
			hidden_effect = {
				random_system_planet = {
					limit = {
						NOT = { exists = archaeological_site }
						is_planet_class = pc_barren
					}
					save_event_target_as = GF_sf_site_target_2
				}
				set_star_flag = flag_GF_sf_event_system
			}
		}
	}

	option = {
		name = "GF_sf.2.a"
		event_target:GF_sf_site_target_2 = {
			enable_special_project = {
				name = "GF_sf_event_project_2"
				location = this
				owner = root.owner
			}
		}
	}
}

#事件6:生成星系、在机械星球建立特殊项目6
ship_event = {
	id = GF_sf.6
	title = "GF_sf.6.name"
	desc = "GF_sf.6.desc"
	picture = GFX_evt_ruined_system
	show_sound = event_activating_unknown_technology
	location = event_target:GF_sf_site_target_6
	trackable = yes
	is_triggered_only = yes
	immediate = {
		hidden_effect = {
			closest_system = {
				min_steps = 2
				max_steps = 5
				spawn_system = {
					min_distance >= 29
					max_distance <= 31
					min_orientation_angle = 24
					max_orientation_angle = 66
					initializer = GF_sf_system
				}
			}
			last_created_system = {
				star = {
					create_ambient_object = {
						type = "contingency_2"
						location = this
					}
					last_created_ambient_object = {
						set_ambient_object_flag = contingency_system_effect_2
						set_location = {
							target = prev
							distance = 0
							angle = random
						}
					}
					set_crisis_sound = ai_empire_ambient_stage_1
				}
			}
			add_extra_hyperlane_to_spawned_system_effect = yes
		}
	}
	option = {
		name = "GF_sf.6.a"
		event_target:GF_sf_site_target_6 = {
			enable_special_project = {
				name = "GF_sf_event_project_6"
				location = this
				owner = root.owner
			}
		}
	}
}
