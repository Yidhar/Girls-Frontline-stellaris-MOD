citizenship_robot_servitude = {
	is_robot_servant = yes
	potential = {
		exists = from
		from = { is_gestalt = no }
		OR = {
			has_trait = trait_mechanical
			has_trait = GF_trait_doll		# GF 人形特质
		}
	}

	allow = {
		hidden_trigger = { exists = from }
		if = {
			# GF 人形物种
			limit = { has_trait = GF_trait_doll }
			from = {
				OR = {
					has_policy_flag = GF_doll_right_servitude
					has_policy_flag = GF_doll_right_low
				}
			}
		}
		else = {
			# GF 非人形机械物种仍然使用原版判定
			custom_tooltip = {
				fail_text = ROBOTS_HAVE_RIGHTS
				from = {
					NOT = { has_policy_flag = ai_full_rights }
				}
			}
			custom_tooltip = {
				fail_text = ROBOTS_ARE_OUTLAWED
				from = {
					NOT = { has_policy_flag = robots_outlawed }
				}
			}
		}
	}

	ai_will_do = {
		factor = 1
	}
}
