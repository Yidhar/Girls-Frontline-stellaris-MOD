#崔耶拉切换职业
triela_become_governor_sfdoll_effect = {
	optimize_memory
	hidden_effect = {
		event_target:sfdoll_leader_triela = {
			kill_leader = {
				show_notification = no
				fire = yes
			}
		}
		create_leader = {
			class = governor
			species = owner_species
			name = NAME_sfdoll_triela
			skill = 1
			set_age = 20
			immortal = yes
			event_leader = yes	# prevents this leader from leading factions and winning elections
			traits = {
				trait = leader_trait_governor_sfdoll_triela
				trait = leader_trait_righteous
			}
			effect = {
				change_leader_portrait = GFX_sfdoll_triela
				set_leader_flag = sfdoll_leader_triela
				save_global_event_target_as = sfdoll_leader_triela
			}
		}
	}
}

#诺耶露变为飞船
effect_create_fleet_sfdoll_noel = {
	optimize_memory
	event_target:sfdoll_leader_noel = {
		kill_leader = {
			show_notification = no
			fire = yes
		}
	}
	if = {
		limit = {
			exists = event_target:fleet_sfdoll_noel
		}
		delete_fleet = event_target:fleet_sfdoll_noel
	}
	create_fleet = {
		name = Name_fleet_sfdoll_noel
		settings = {
			can_upgrade = yes
			can_change_composition = yes
			can_change_leader = no
			uses_naval_capacity = no
			spawn_debris = no
			is_boss = no
			can_disband = yes
		}
		effect = {
			save_global_event_target_as = fleet_sfdoll_noel
			set_owner = root
			set_location = root.capital_scope
			create_ship = {
				name =  Name_ship_sfdoll_noel
				design = design_ship_sfdoll_noel
				prefix = no
				upgradable = no
				effect = {	save_global_event_target_as = ship_sfdoll_noel	}
			}
		}
	}
}
